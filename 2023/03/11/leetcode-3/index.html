class Solution {
	/**
	 * 我们遍历数组，跟踪我们目前看到的字母数量和数字数量之间的差异。如果我们看到之前看到的差异，我们就知道之前出现的差异与当前元素之间的子数组具有相同数量的字母和数字。我们跟踪迄今为止我们见过的最长的此类子阵列
	 *
	 * @param array 输入数组。
	 * @return 字母和数字数目相等的最长子数组。
	 */
	public String[] findLongestSubarray(String[] array) {
		var pre = 0;
		var len = 0;
		var count_Letters = 0;
		var count_nums = 0;
		int[] difference = new int[array.length << 1];
		for (int i = 0; i < array.length; ++i) {
			//负数由0至array.length-1记录
			int idx = count_nums < count_Letters ?
					count_Letters - count_nums - 1 : (count_nums - count_Letters) + array.length;//正数或0 由
			// array.length至((array.length<<1)-1)记录
			difference[idx] = (difference[idx] == 0) ? i + 1 : difference[idx];
			if (Character.isLetter( array[i].charAt( 0 ) )) ++count_Letters;
			else ++count_nums;
			var diff = count_nums < count_Letters ?
					count_Letters - count_nums - 1 : (count_nums - count_Letters) + array.length;
			int integer = difference[diff];
			if (integer != 0 && i - integer + 2 > len) {
				len = i - integer + 2;
				pre = integer - 1;
			}
		}
		return Arrays.copyOfRange( array,
		                           pre,
		                           len + pre );
	}
}

给定行和列的和求可行矩阵link
class Solution {

  /**
   * *|MARKER_CURSOR|*
   *
   * @param rowSum [3,8]
   * @param colSum [3,8]
   */
  public int[][] restoreMatrix(int[] rowSum, int[] colSum) {
    var m = rowSum.length;
    var n = colSum.length;
    //当前最大值取决于能保证后续可以赋值
    var row = 0;
    var col = new int[n];
    var ans = new int[m][n];
    for (int i = 0; i < m; ++i) {
      for (int j = 0; j < n; ++j) {
        int min = Math.min(rowSum[i] - row, colSum[j] - col[j]);
        ans[i][j] = min;
        row += min;
        col[j] += min;
      }
      row = 0;
    }
    return ans;
  }
}

最大网络秩link
class Solution {

	/**
	* 我们创建一个矩阵，其中每一行代表一个城市，每一列代表一个城市。每个单元格的值是行代表的城市和列代表的城市之间的道路数量。最后一列是该行所代表的城市的索引。最后一行是该行所代表的城市连接的道路数量
	*
	* @param n     城市数量
	* @param roads 道路列表，其中 roads[i] = [cityAi, cityBi] 表示 cityAi 和 cityBi 之间存在一条道路。
	* @return 两个城市之间可以建造的最大道路数量。
	*/
	public int maximalNetworkRank(int n, int[][] roads) {
		var matrix = new int[n][n + 1];
		for (int[] road : roads) {
			++matrix[road[0]][road[1]];
			++matrix[road[0]][n];
			++matrix[road[1]][road[0]];
			++matrix[road[1]][n];
		}
		int ans = 0;
		for (int i = 0; i < matrix.length; ++i) {
			for (int j = i + 1; j < matrix.length; ++j) {
				var cur = matrix[i][n] + matrix[j][n] - (matrix[i][j] == 0 ? 0 : 1);
				ans = Math.max(ans, cur);
			}
		}
		return ans;
	}
}
