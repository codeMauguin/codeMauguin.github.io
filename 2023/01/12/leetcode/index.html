    public String evaluate(String s, List<List<String>> knowledge) {
        if (s == null || s.isEmpty()) return s;
        Map<String, String> knowledges = knowledge
            .stream()
            .collect(Collectors.toMap(strings -> strings.get(0), strings -> strings.get(1)));
        StringBuilder builder = new StringBuilder();
        StringBuilder key = new StringBuilder();
        int index = 0;
        char[] chars=s.toCharArray();
        while (index < s.length()) {
            char target = chars[index++];
            if (target != '(') {
                builder.append(target);
            } else {
                while (index < s.length()) {
                    char charAt = chars[index++];
                    if (charAt == ')') {
                        builder.append(knowledges.getOrDefault(key.toString(), "?"));
                        key.setLength(0);
                        break;
                    } else {
                        key.append(charAt);
                    }
                }
            }
        }
        return builder.toString();
    }

    function evaluate(s: string, knowledge: string[][]): string {
    const map: Record<string, string> = {};
    knowledge.forEach(value => map[value[0]] = value[1]);
    return s.replace(/\(.*?\)/g, (substring) => {
        return map[substring.slice(1, -1)] ?? "?";
    });
}

public int rearrangeCharacters(String s, String target) {
        if (s == null || target == null || s.isEmpty() || target.isEmpty()) return 0;
        char[] chars = target.toCharArray();
        Map<Character, Integer> map = new HashMap<>();
        for (char aChar : chars) {
            map.put(aChar, map.getOrDefault(aChar, 0) + 1);
        }
        char[] charArray = s.toCharArray();
        Map<Character, Integer> hashMap = new HashMap<>();
        for (char c : charArray) {
            hashMap.put(c, hashMap.getOrDefault(c, 0) + 1);
        }
        int ans = Integer.MAX_VALUE;
        for (Map.Entry<Character, Integer> entry : map.entrySet()) {
            Character character = entry.getKey();
            Integer integer = entry.getValue();
            ans = Math.min(ans, Math.floorDiv(hashMap.getOrDefault(character, 0), integer));
        }
        return ans;
    }

public int rearrangeCharacters(String s, String target) {

        char[] chars = target.toCharArray();
        Map<Character, Long> map = IntStream.range(0, chars.length).mapToObj(value -> chars[value])
            .collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));
        char[] charArray = s.toCharArray();
        Map<Character, Long> hashMap =
        IntStream.range(0, charArray.length).mapToObj(value -> charArray[value])
            .collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));
        int ans = Integer.MAX_VALUE;
        for (Map.Entry<Character, Long> entry : map.entrySet()) {
            Character character = entry.getKey();
            Long integer = entry.getValue();
            ans = Math.toIntExact(Math.min(ans, Math.floorDiv(hashMap.getOrDefault(character, 0L), integer)));
        }
        return ans;
    }

public int minMaxGame(int[] nums) {
   if (nums.length == 1) return nums[0];
   int[] array = IntStream
     .range(0, nums.length / 2)
     .map(operand ->
       ((operand & 1) == 1)
         ? Math.max(nums[operand * 2], nums[operand * 2 + 1])
         : Math.min(nums[operand * 2], nums[operand * 2 + 1])
     )
     .toArray();

   return minMaxGame(array);
 }

public boolean areSentencesSimilar(String sentence1, String sentence2) {
    if (sentence1.equals(sentence2)) return true;
    String[] split1 = sentence1.split(" ");
    String[] split2 = sentence2.split(" ");
    int start1 = 0;
    int start2 = 0;
    int end1 = split1.length - 1;
    int end2 = split2.length - 1;
    while (start1 < split1.length && start2 < split2.length) {
      if (split1[start1].equals(split2[start2])) {
        start1++;
        start2++;
      } else break;
    }
    if (start1 == split1.length || start2 == split2.length) return true;
    while (end1 >= 0 && end2 >= 0) {
      if (split1[end1].equals(split2[end2])) {
        end2--;
        end1--;
      } else break;
    }
    return end1 == -1 || end2 == -1 || end1 < start1 || end2 < start2;
  }

private int rev(int num) {
		int a = 0;
		while (num != 0) {
			a = a * 10 + num % 10;
			num /= 10;
		}
		return a;
	}

求出 MK 平均值link
import java.util.ArrayList;
import java.util.Comparator;
import java.util.Iterator;
import java.util.Map.Entry;
import java.util.TreeMap;

class MKAverage {

  int m;
  int k;
  ArrayList<Integer> list = new ArrayList<>();
  TreeMap<Integer, Integer> treeMap = new TreeMap<>(Comparator.reverseOrder());
  int cur = 0;
  int start = 0;
  boolean f = false;

  public MKAverage(int m, int k) {
    this.m = m;
    this.k = k;
    f = 2 * k >= m;
  }

  public void addElement(int num) {
    if (list.size() >= m) {
      Integer removeFirst = list.get(start++);
      cur -= removeFirst;
      Integer merge = treeMap.merge(removeFirst, -1, Integer::sum);
      if (merge == 0) treeMap.remove(removeFirst);
    }
    list.add(num);
    treeMap.merge(num, 1, Integer::sum);
    cur += num;
  }

  public int calculateMKAverage() {
    if (list.size() < m) return -1;
    if (f) return 0;
    Iterator<Entry<Integer, Integer>> iterator = treeMap.entrySet().iterator();
    int prev = 0;
    int next = 0;
    int ans = 0;
    Integer key = null;
    int total = m - k - k;
    while (true) {
      Entry<Integer, Integer> el = iterator.next();
      key = el.getKey();
      Integer integer = el.getValue();
      if (prev + integer >= k) {
        int size = integer - (k - prev);
        if (size > 0) {
          if (size >= total) {
            ans += key * total;
            return (int) (ans / total);
          } else {
            next = size;
            ans += key * size;
          }
        }
        break;
      } else {
        prev += integer;
      }
    }
    while (true) {
      Entry<Integer, Integer> el = iterator.next();
      key = el.getKey();
      Integer integer = el.getValue();
      if (next + integer >= total) {
        ans += key * (total - next);
        next = total;
        return (int) (ans / total);
      } else {
        next += integer;
        ans += integer * key;
      }
    }
  }
}

 * MKAverage obj = new MKAverage(m, k);
 * obj.addElement(num);
 * int param_2 = obj.calculateMKAverage();
 */

class Solution {

  /**
   * @param password
   * @return
   */
  public boolean strongPasswordCheckerII(String password) {
    if (password.length() < 8) return false;
    boolean hasX = false;
    boolean hasD = false;
    boolean hasS = false;
    boolean hasT = false;
    char[] charArray = password.toCharArray();
    int i = 0;
    while (i < charArray.length) {
      char c = charArray[i];
      if (i > 0 && charArray[i - 1] == charArray[i]) return false;
      if (Character.isLowerCase(c)) {
        hasX = true;
      } else if (Character.isUpperCase(c)) {
        hasD = true;
      } else if (Character.isDigit(c)) {
        hasS = true;
      } else if ("!@#$%^&*()-+".indexOf(c) > -1) {
        hasT = true;
      }
      i++;
    }
    return hasX && hasD && hasS && hasT;
  }
}

import java.util.function.Predicate;
import java.util.regex.Pattern;

class Solution {

  Predicate<String> predicate = Pattern
    .compile(
      "^(?=.*\\d)(?=.*[a-z])(?=.*[A-Z])(?=.*[!@#$%^&*()\\-+])(?!.*(.)\\1+).{8,}$"
    )
    .asMatchPredicate();

  public boolean strongPasswordCheckerII(String password) {
    return predicate.test(password);
  }
}

class Solution {
    public int[] findingUsersActiveMinutes(int[][] logs, int k) {
        Map<Integer, Set<Integer>> activeMinutes = new HashMap<Integer, Set<Integer>>();
        for (int[] log : logs) {
            int id = log[0], time = log[1];
            activeMinutes.putIfAbsent(id, new HashSet<Integer>());
            activeMinutes.get(id).add(time);
        }
        int[] answer = new int[k];
        for (Set<Integer> minutes : activeMinutes.values()) {
            int activeCount = minutes.size();
            answer[activeCount - 1]++;
        }
        return answer;
    }
}

class Solution {
   public int minSideJumps(int[] obstacles) {
      int[][] dp = new int[3][obstacles.length];
      int max = obstacles.length;
      dp[0][0] = obstacles[0] == 1 ? max : 1;
      dp[1][0] = obstacles[0] == 2 ? max : 0;//没有侧跳
      dp[2][0] = obstacles[0] == 3 ? max : 1;
      for (int i = 1; i < obstacles.length; ++i) {
         for (int j = 0; j < 3; ++j) {
            dp[j][i] = dp[j][i - 1];
         }
         if (obstacles[i] > 0) {
            //当前存在障碍物
             if (obstacles[i - 1] == obstacles[i]) {//前一个障碍物和当前一致，其他两道路直接前行
				dp[obstacles[i] - 1][i] = max;
				continue;
			}
            int a = obstacles[i] == 2 || obstacles[i] == 3 ? 0 : 1;
            int b = obstacles[i] == 3 ? 1 : 2;
            jump(obstacles,
                 dp,
                 i,
                 b,
                 a);
            jump(obstacles,
                 dp,
                 i,
                 a,
                 b);
            dp[obstacles[i] - 1][i] = max;
         }
      }
      return Math.min(Math.min(dp[0][obstacles.length - 1],
                               dp[1][obstacles.length - 1]),
                      dp[2][obstacles.length - 1]);
   }
   
   /**
    * 跳 判断前一个点是否为当前障碍物
    *
    * @param obstacles 障碍
    * @param dp        dp
    * @param i         我
    * @param a         一个
    * @param b         b
    */
   private void jump(final int[] obstacles,
                     final int[][] dp,
                     final int i,
                     final int a,
                     final int b) {
      if (obstacles[i - 1] - 1 == b) {
         dp[b][i] = dp[a][i - 1] + 1;
      } else {
         dp[b][i] = Math.min(dp[b][i - 1],
                             dp[obstacles[i] - 1][i - 1] + 1);
      }
   }
}
