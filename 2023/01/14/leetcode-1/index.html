设计数字容器系统

设计数字容器系统link
class NumberContainers {

    private final Map<Integer, int[]> index = new HashMap<>();
    private final Map<Integer, PriorityQueue<Integer>> values = new HashMap<>();

    private final Map<Integer, Map<Integer, Integer>> flags = new HashMap<>();

    public NumberContainers() {

    }

    public void change(int index, int number) {
        int[] ints = this.index.computeIfAbsent(index, v -> new int[1]);
        if (ints[0] != 0) {
            //标记删除
            Map<Integer, Integer> integerIntegerMap = flags.computeIfAbsent(ints[0], integer -> new HashMap<>());
            integerIntegerMap.put(index, 1);
        }
        ints[0] = number;
        values.computeIfAbsent(number, v -> new PriorityQueue<>()).offer(index);
        Map<Integer, Integer> integerIntegerMap = flags.computeIfAbsent(ints[0], integer -> new HashMap<>());
        //防止之前被标记删除
        integerIntegerMap.put(index, 0);
    }

    public int find(int number) {
        PriorityQueue<Integer> integers = values.get(number);
        if (integers == null || integers.isEmpty()) return -1;
        while (!integers.isEmpty()) {
            Integer peek = integers.peek();
            Map<Integer, Integer> integerIntegerMap = flags.get(number);
            //校验是否已经被删除
            if (integerIntegerMap.get(peek) == 1) {
                integers.poll();//移除当前过期的下标
            } else {
                return peek;
            }
        }
        return -1;
    }
}

 * Your NumberContainers object will be instantiated and called as such:
 * NumberContainers obj = new NumberContainers();
 * obj.change(index,number);
 * int param_2 = obj.find(number);
 */

    字符串的前缀分数和link
    class word {

  int count = 0;
  word[] child = new word[26];
}

class Solution {

  private int index(char target) {
    return target - 'a';
  }

  private void builder(word[] parent, char[] target, int index) {
    if (index >= target.length) return;
    int idx = index(target[index]);
    if (parent[idx] == null) {
      parent[idx] = new word();
    }
    ++parent[idx].count;
    builder(parent[idx].child, target, index + 1);
  }

  private int build(word[] parent, char[] target, int index) {
    if (index >= target.length) return 0;
    int idx = index(target[index]);
    return parent[idx].count + build(parent[idx].child, target, index + 1);
  }

  public int[] sumPrefixScores(String[] words) {
    word[] ws = new word[26];
    List<char[]> aList = new ArrayList<>(words.length);
    for (String word : words) {
      char[] charArray = word.toCharArray();
      aList.add(charArray);
      builder(ws, charArray, 0);
    }
    var ans = new int[words.length];
    for (int i = 0; i < words.length; ++i) {
      ans[i] = build(ws, aList.get(i), 0);
    }
    return ans;
  }
}

从给定原材料中找到所有可以做出的菜

中等

  • 先做出菜并加入材料中

  • 循环做剩下的菜，如果做不出来直接结束

  • 将已经有的原材料删除不做下次循环，优化时间

从给定原材料中找到所有可以做出的菜link
class Solution {

    public List<String> findAllRecipes(
            String[] recipes,
            List<List<String>> ingredients,
            String[] supplies
    ) {
        Map<String, Boolean> sup = Arrays
                .stream(supplies)
                .collect(
                        Collectors.toMap(supplier -> supplier, supplier -> true, (a, b) -> b)
                );
        Map<String, Integer> task = IntStream
                .range(0, recipes.length)
                .boxed()
                .collect(Collectors.toMap(i -> recipes[i], i -> i, (a, b) -> b));
        List<String> a = new ArrayList<>();
        int count = 0;
        do {
            count = 0;
            Map<String, Integer> c = new HashMap<>();
            for (Map.Entry<String, Integer> stringIntegerEntry : task.entrySet()) {
                Integer value = stringIntegerEntry.getValue();
                List<String> strings = ingredients.get(value);
                //检查string 是否都有
                boolean b = true;
                for (int i = 0, stringsSize = strings.size(); i < stringsSize; ++i) {
                    String string = strings.get(i);
                    if (!sup.containsKey(string)) {
                        b = false;
                        ingredients.set(value, strings.subList(i, strings.size()));
                        break;
                    }
                }
                if (b) {
                    sup.put(stringIntegerEntry.getKey(), true);
                    ++count;
                    a.add(stringIntegerEntry.getKey());
                } else {
                    c.put(stringIntegerEntry.getKey(), stringIntegerEntry.getValue());
                }
            }
            task = c;
        } while (count > 0 && !task.isEmpty());
        return a;
    }
}

会议室 III

困难题

  • 使用大小堆队列取编号最小的空闲会议室
会议室 IIIlink
	class Solution {
	/**
	 * 给定会议室数量和会议列表，返回会议最多的房间
	 *
	 * @param n        建筑物的房间数
	 * @param meetings 一个二维整数数组，其中每一行代表一个会议。每行的第一个元素是会议的开始时间，第二个元素是会议的结束时间。
	 */
	public int mostBooked(int n,
	                      int[][] meetings) {
		//选择编号最小的会议室
		PriorityQueue<long[]> task = new PriorityQueue<>( (a, b) -> a[2] == b[2] ? Long.compare( a[0],
		                                                                                           b[0] ) :
				Long.compare( a[2],
		                                                                                                                     b[2] ) );
		Arrays.sort( meetings,
		             Comparator.comparingInt( m -> m[0] ) );//已会议室开始时间排序，优先时间早的进行开始会议
		PriorityQueue<long[]> meets = new PriorityQueue<>( Comparator.comparingLong( value -> value[0] ) );
		IntStream.range( 0,
		                 n )
				.mapToObj( i -> new long[]{i, 0, 0} )
				.forEach( meets::offer );
		var idx = Long.MAX_VALUE;
		var max = Long.MIN_VALUE;
		var cur = 0L;
		for (int[] meeting : meetings) {
			//找出会议室
			//先将之前可能已经结束的任务拿到会议室里
			if (meets.isEmpty( )) {
				//当前会议室被用完，调节时间到最快结束任务的会议
				cur = task.peek( )[2];
			}
			cur = Math.max( meeting[0],
			                cur );//将时间拉到任务时间
			while (!task.isEmpty( ) && task.peek( )[2] <= cur) {//将会议已经结束的推到正常的会议室中
				meets.offer( task.poll( ) );
			}
			long[] poll = meets.poll( );
			++poll[1];//标记会议室使用次数
			poll[2] = (cur + meeting[1] - meeting[0]);//标记当前会议室结束时间
			task.offer( poll );//将放入等待队列
			if (poll[1] > max) {//记录最大次数的编号
				max = poll[1];
				idx = poll[0];
			} else if (poll[1] == max) {//记录同次数最小的编号
				idx = Math.min( idx,
				                poll[0] );
			}
		}
		return (int) idx;
	}
}

模糊坐标

  • Stream流
  • 切割字符串，在重新组合
[816. 模糊坐标 - 力扣（Leetcode）](https://leetcode.cn/problems/ambiguous-coordinates/description/)
class Solution {

  Predicate<String> pattern = Pattern.compile("\\d*0$").asMatchPredicate();
  String template = "(%s, %s)";

  /**
   * “给定两个字符串列表，返回一个字符串列表，该列表是将第一个列表的每个元素与第二个列表的每个元素合并的结果。”
   *
   * 该函数是用 Java 编写的，但不是很地道。它不是很可读，也不是很简洁
   *
   * @param a [1, 2, 3]
   * @param b [1, 2, 3]
   * @return 字符串列表。
   */
  private List<String> merge(List<String> a, List<String> b) {
    return a
      .stream()
      .flatMap(v -> {
        return b.stream().map(c -> String.format(template, v, c));
      })
      .toList();
  }

  /**
   * 它将字符串分成两部分，第一部分是整数部分，第二部分是小数部分
   *
   * @param str 要拆分的字符串
   * @return 字符串列表。
   */
  private List<String> spit(String str) {
    //将字符切割成
    if (str.length() == 1) {
      return List.of(str);
    }
    List<String> strings = new ArrayList<>();
    boolean isZero = str.charAt(0) == '0'; //检查是否是0开头的
    if (!isZero) strings.add(str);
    //如果以0结尾则不能分割
    if (pattern.test(str)) return strings;
    if (isZero) {
      //以0开头
      if (str.length() != 1) {
        strings.add("0." + str.substring(1));
      }
      return strings;
    }
    for (int i = 1; i < str.length(); ++i) {
      strings.add(str.substring(0, i) + "." + str.substring(i));
    }
    return strings;
  }

  /**
   * 对于字符串的每个可能拆分，我们采用左右两半的可能拆分，并以所有可能的方式组合它们
   *
   * @param s “（0123）”
   * @return 字符串列表。
   */
  public List<String> ambiguousCoordinates(String s) {
    List<String> ans = new ArrayList<>();
    s = s.substring(1);
    s = s.substring(0, s.length() - 1);
    for (int i = 1; i < s.length(); ++i) {
      List<String> merge = merge(spit(s.substring(0, i)), spit(s.substring(i)));
      ans.addAll(merge);
    }
    return ans;
  }
}
